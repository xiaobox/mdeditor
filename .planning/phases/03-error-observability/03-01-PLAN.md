// createModuleLogger('Name') produces a logger with prefix '[Name]'.
// When log.warn('message', err) is called, console.warn receives:
//   arg[0]: '[HH:MM:SS] [Name]'   (timestamp + module prefix)
//   arg[1]: 'message'              (operation description)
//   arg[2]: Error                  (error object)
//
// In Vitest (import.meta.env.DEV = true), logger delegates directly
// to console.* -- so vi.spyOn(console, 'warn') captures the calls.

export function createModuleLogger(moduleName) {
  return createLogger(`[${moduleName}]`)
}
// Returns: { debug, info, warn, error, group, groupEnd }
